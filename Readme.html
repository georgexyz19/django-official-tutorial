Part 7 Customize Admin Form

The code below is interesting.

class ChoiceInline(admin.StackedInline):
    model = Choice
    extra = 3
   
class QuestionAdmin(admin.ModelAdmin):
    fieldsets = ...
    inlines = [ChoiceInline]
    
admin.site.register(Question, QuestionAdmin)
